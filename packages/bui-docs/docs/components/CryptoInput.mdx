# CryptoInput

## Basic Example

The Crypto Input component is used for simplifying token selections in a Dapp

export const CryptoInputComponentExample = function () {
  const [state, setVal] = useState("");
  const [selectedToken, setSelectedToken] = useState(CryptoList[0]);
  return (
    <CryptoInput
      selectedToken={selectedToken}
      listOfCurrencies={CryptoList}
      onChange={setVal}
      value={state}
      onSelectToken={setSelectedToken}
      price="1"
    />
  );
};

<DisplayExample row justifyContent="center">
  <CryptoInputComponentExample />
</DisplayExample>
